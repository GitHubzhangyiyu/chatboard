CREATE DATABASE messageboard CHARACTER SET utf8;
USE messageboard;
CREATE TABLE users(
user_id MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT,
username VARCHAR(30) NOT NULL,
pass CHAR(40) NOT NULL,
email VARCHAR(60) NOT NULL,
permission BOOLEAN NOT NULL DEFAULT FALSE,
PRIMARY KEY (user_id),
UNIQUE (username),
UNIQUE (email),
INDEX login (username, pass)
);
CREATE TABLE messages(
message_id MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT,
from_uid MEDIUMINT UNSIGNED NOT NULL,
to_uid MEDIUMINT UNSIGNED NOT NULL,
root_mid MEDIUMINT UNSIGNED NOT NULL,
status BOOLEAN NOT NULL DEFAULT TRUE,
text VARCHAR(128) NOT NULL,
addtime TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
PRIMARY KEY (message_id)
);
